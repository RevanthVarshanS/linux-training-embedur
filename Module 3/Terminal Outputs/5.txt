#!/bin/bash

SOURCE="$1"
BACKUP="$2"
EXT="$3"

if [ ! -d "$SOURCE" ]; then
    echo "Source does not exist"
    exit 1
fi

if [ ! -d "$BACKUP" ]; then
    mkdir -p "$BACKUP" || { echo "Failed to create backup"; exit 1; }
fi

FILES=("$SOURCE"/*"$EXT")

if [ ${#FILES[@]} -eq 0 ]; then
    echo "No matching files found"
    exit 0
fi

for file in "${FILES[@]}"
do
    cp "$file" "$BACKUP"
done

echo "Backup Done"
